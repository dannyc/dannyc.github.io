What is a graph?

One implementation


function traverseGraph(firstNodeId, allNodes){
  var visitedIds = []; // keeps tracks of nodes that we have already traversed
  var queue = allNodes[firstNodeId].associated; //initialize the queue with the first node we are going to traverse
  while(queue && queue.length > 0){ //loop until we've traversed the whole double booked graph
  	var nodeId = queue.shift(); //take the first item out of our queue
  	if(visitedIds.indexOf(nodeId) < 0){ //have we NOT gone down this route already
  	  visitedIds.push(nodeId); //we are now going down it- so add it to the list of visited
  	  var nextIds = allNodes[nodeId].associated; //get the next node associated ids
  	  //continue to traverse the graph by adding to queue more node ids if needed
  	  for(var i = 0,l = nextIds.length; i < l; i++){ //loop through this nodes associated nodes
  	    var nextId = nextIds[i];
  	    if(visitedIds.indexOf(nextId) < 0 && queue.indexOf(nextId) < 0){ //have we NOT gone down this route already, and not already in the queue
  		  queue.push(nextId); //add it to the queue
  	    }
  	  }
    }
  }
  return visitedIds; //list of all the nodes that we've been to already
}
